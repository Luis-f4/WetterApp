let player;
let jumpForce = 0;
let movement = 0;
let ground;
let platforms = [];
let coin;

let platformDirection = 1;
let coinDirection = 1;

function setup() {
    createCanvas(1250, 550);
    resetGame();
}

function resetGame() {
    player = createVector(width * 0.5, height * 0.85); // Spielerposition
    platforms = [
        createVector(width * 0.3, height * 0.85),  // Plattform 1
        createVector(width * 0.52, height * 0.65), // Plattform 2
        createVector(width * 0.8, height * 0.45)   // Plattform 3
    ];
    coin = createVector(width * 0.96, height * 0.36); // Münzposition
    ground = height * 0.9; // Bodenhöhe

    document.getElementById("h1").style.opacity = 0;
    document.getElementById("btn").style.display = "none";
}

function restartGame() {
    resetGame();
    loop();
}

function drawGame() {
    background('DeepSkyBlue');
    rectMode(CENTER);

    // Spieler zeichnen
    fill('white');
    rect(player.x, player.y, width * 0.03, height * 0.06); // Spielergröße angepasst

    // Plattformen zeichnen
    fill('green');
    for (let platform of platforms) {
        rect(platform.x, platform.y, width * 0.12, height * 0.036); // Plattformgröße
    }

    // Coin zeichnen
    if (coin) {
        fill('gold');
        ellipse(coin.x, coin.y, width * 0.024, height * 0.043); // Münze
    }

    addNewPlatform();
}

function draw() {
    drawGame();

    // Bewegung und Schwerkraft
    player.y += jumpForce;
    player.x += movement;

    let onGround = false;

    // Überprüfung: Boden oder Plattformen
    if (player.y >= ground) {
        onGround = true;
        player.y = ground;
    } else {
        for (let platform of platforms) {
            if (
                player.x + width * 0.015 >= platform.x - width * 0.06 && // Breite der Plattform
                player.x - width * 0.015 <= platform.x + width * 0.06 &&
                Math.abs(player.y + height * 0.03 - (platform.y - height * 0.018)) <= 1 // Stabilisierung
            ) {
                onGround = true;
                player.y = platform.y - height * 0.06; // Spieler stabil auf Plattform
                break;
            }
        }
    }

    // Bewegliche Plattform
    platforms[2].x += platformDirection * (width * 0.0016);

    if (platforms[2].x <= width * 0.5) {
        platformDirection = 1;
    } else if (platforms[2].x >= width * 0.72) {
        platformDirection = -1;
    }

    // Bewegliche Münze
    if (coin.y <= height * 0.27) {
        coinDirection = 1;
    } else if (coin.y >= height * 0.45) {
        coinDirection = -1;
    }
