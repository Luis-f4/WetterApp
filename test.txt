function checkAblaufdatum(heutigesDatum, ablaufDatum) {
    // Konvertiere die Strings in Date-Objekte
    const heute = new Date(heutigesDatum);
    const ablauf = new Date(ablaufDatum);
    
    // Berechne die Differenz in Monaten
    const differenzInMonaten = (ablauf.getFullYear() - heute.getFullYear()) * 12 + (ablauf.getMonth() - heute.getMonth());

    if (heute > ablauf) {
        console.log("Abgelaufen");
    } else if (differenzInMonaten < 3) {
        console.log("Abgelaufen");
    } else {
        console.log("Alles in Ordnung");
    }
}

// Beispielaufruf:
checkAblaufdatum("2024/10/18", "2025/01/17"); // Sollte "Abgelaufen" ausgeben
