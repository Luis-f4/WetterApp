let player; // 360 | 455
let jumpForce = 0;
let movement = 0;
let ground;
let platform;
let platform2;

function setup() {
    createCanvas(1900, 900);
    player = createVector(width / 2, height - 50);
    platform = createVector(100, 800);
    platform2 = createVector(400, 650);
    ground = height - 50;
}

function draw() {
    background('DeepSkyBlue');
    rectMode(CENTER);
    fill('white');
    rect(player.x, player.y, 50, 50);
    fill('green');
    rect(platform.x, platform.y, 150, 20);
    rect(platform2.x, platform2.y, 150, 20);


    player.y += jumpForce;
    player.x += movement;

    if (player.y >= ground) {
        jumpForce = 0;
    } else if (
        (player.x + 25 >= platform.x - 75 && player.x - 25 <= platform.x + 75 && player.y + 25 >= platform.y - 10 &&
player.y + 25 <= platform.y + 10)
        

    ) {
        jumpForce = 0;
        player.y = platform.y - 35;
    } else if(        player.x + 25 >= platform2.x - 75 &&
        player.x - 25 <= platform2.x + 75 &&
        player.y + 25 >= platform2.y - 10 &&
        player.y + 25 <= platform2.y + 10){
            jumpForce = 0;
            player.y = platform2.y - 35;
        
    }else {
        jumpForce += 0.5;
    }

    if (keyIsDown(32) && (player.y >= ground || player.y == platform.y - 35)) {
        jumpForce = -12;
    }

    if (keyIsDown(32) && (player.y >= ground || player.y == platform2.y - 35)) {
        jumpForce = -12;
    }


    if (keyIsDown(LEFT_ARROW)) {
        movement = -10;
    } else if (keyIsDown(RIGHT_ARROW)) {
        movement = +10;
    } else {
        movement = 0;
    }


}
